game_logic_controller.vhd
entity game_logic_controller is
    generic
    (
        INIT_LIFE: integer := 5
        MAX_CROW: integer := 8
        MAX_SHIT_PER_CROW: integer := 4
    );
    port
    (
        RST: in std_logic;
    	NEW_GAME: in std_logic;
    	CLK: in std_logic;
    	RANDOM_SEED: in std_logic_vector(15 downto 0);

    	ROLL: in std_logic_vector(15 downto 0);
        PITCH: in std_logic_vector(15 downto 0);
    	
    	GAME_STATE: out std_logic_vector(1 downto 0); --00 未开始 -01进行中 --10已结束
    	PLAYER_POS: out std_logic_vector(9 downto 0);
    	REMAIN_LIFE: out std_logic_vector(2 downto 0);
    	SCORE: out std_logic_vector(19 downto 0);
    	CROW_IN_SCREEN: out std_logic_vector(7 downto 0); --1 bit per crow
    	CROW_ROW: out std_logic_vector(15 downto 0); -- 2 bit per crow  00,10: left->right  01,11: right->left
    	CROW_POS: out std_logic_vector(79 downto 0); -- 10 bit per crow
    	HAS_SHIT: out std_logic_vector(31 downto 0);
    	SHIT_HEIGHT: out std_logic_vector(319 downto 0); -- 10 bit per shit
    );
end;
